# 表格布局

> 表格布局的实现较为复杂，正在开发中……

```vue
<template>
  <div>
    <v-schema-form
        v-model:value="value"
        :schema="schema"/>
  </div>
</template>
<script lang="ts" setup>
  import {ref} from 'vue';
  import {registerAntd} from '../../../schema-form';
  import {SchemaFormField} from '../../../../types';

  registerAntd();
  const value = ref([{}])
  const schema: SchemaFormField = {
    props: {
      labelWidth: 100,
      labelPosition: 'left'
    },
    array: true,
    arrayComponent: 'table',
    fields: {
      name: {
        title: '0名称',
        type: 'string'
      },
      $grid: {
        type: 'grid',
        layout: [12, 12],
        props: {
          gutter: 20
        },
        fields: {
          other: {
            title: '其他',
            type: 'string'
          },
          other2: {
            title: '其他2',
            type: 'string'
          }
        }
      }
    }
  }
</script>
```
