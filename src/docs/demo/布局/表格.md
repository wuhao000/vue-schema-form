# 表单列表

```vue
<template>
  <a-layout>
    <a-layout-content>
      <v-schema-form
          v-model:value="value"
          :schema="schema"
          class="demo-form"
          @ok="onOk"/>
      <show-value :value="value3"/>
    </a-layout-content>
  </a-layout>
</template>
<script lang="ts">
  import {registerAntd} from '../../../schema-form';
  import {SchemaFormField} from '../../../../types';
  import {defineComponent, ref} from 'vue';
  
  registerAntd();
  export default defineComponent({
    name: 'Demo',
    setup() {
      const value = ref([{
        basic: {
          aa: '1234',
          cc: {
            cc: '11'
          }
        }
      }]);
      const value2 = ref([{
        a: '1',
        b: '2'
      }]);
      const value3 = ref({
        name: ['1','2']
      })
      return {
        value3,
        value2,
        schema: {
          props: {
            title: '对象列表',
            labelWidth: '120px',
            labelPosition: 'left'
          },
          type: 'object',
          array: true,
          arrayComponent: 'table',
          title: '对象列表',
          arrayProps: {
            addText: '这是定制的添加文案',
            removeText: '这是定制的删除文案',
            maxItems: 3
          },
          fields: {
            aa: {
              type: 'string',
              title: '字段1',
              required: true,
              description: 'cccdaffew威锋网安抚多少安发答案水电费撒粉奥数范德萨'
            },
            bb: {
              type: 'string',
              title: '字段2',
              description: 'aaa'
            },
            cc: {
              type: 'select',
              title: '选择字段'
            }
          }
        } as SchemaFormField,
        value,
        onOk() {
          console.log(value.value);
        }
      };
    }
  });
</script>
```
