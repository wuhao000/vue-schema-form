# 表单组件 DForm

## 属性说明

属性|说明|类型|默认值
---|---|---|---
labelWidth|标签宽度|number,string|无
disabled|禁用表单（禁止表单内所有输入组件）|boolean|无


```vue
<template>
  <div>
    <d-form label-width="120px">
      <d-form-item label="aaa">
        <a-input/>
      </d-form-item>
      <d-form-item
          label="bbb"
          :label-col="{span: 8}"
          :wrapper-col="{span: 16}">
        <a-input/>
      </d-form-item>
      <d-form-item
          label="ccc"
          label-width="180px">
        <a-input/>
      </d-form-item>
    </d-form>
    <d-form>
      <d-form-item label="aaa">
        <a-input/>
      </d-form-item>
    </d-form>
    <a-form>
      <a-form-item label="bbb">
        <a-input/>
      </a-form-item>
    </a-form>
  </div>
</template>
<script lang="ts">
  import {defineComponent, ref} from 'vue';
  import {registerAntd, FieldTypes} from '../../schema-form';

  registerAntd();
  export default defineComponent({
    name: 'Demo',
    setup() {
      const value = ref();
      const rangeValue = ref([]);
      const rangeValue2 = ref(null);
      const rangeValue3 = ref({
        timerange: ['12:02-12:44', '13:44-15:44']
      });
      return {
        value,
        rangeValue,
        rangeValue2,
        rangeValue3,
        schema: {
          fields: {
            image: {
              type: FieldTypes.Picture,
              props: {
                action: '/api/business_entity/file/upload',
                data: {
                  access_token: '90fa61ca-1e96-42ed-ba91-5ae2c2e1ed20'
                }
              }
            },
            file: {
              type: FieldTypes.File,
              props: {
                action: '/api/business_entity/file/upload',
                data: {
                  access_token: '90fa61ca-1e96-42ed-ba91-5ae2c2e1ed20'
                }
              }
            },
            timerange: {
              title: 'time',
              type: 'timerange',
              array: true,
              props: {
                format: 'HH:mm',
                valueType: 'string'
              }
            }
          }
        }
      };
    }
  });
</script>
```
